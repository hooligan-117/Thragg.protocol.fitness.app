Supabase Setup Guide

Add a supabase.js file to your GitHub repo with this structure:

// supabase.js
// Add this script tag to index.html BEFORE closing </body>:
// <script src="supabase.js"></script>

const SUPABASE_URL = 'https://YOUR_PROJECT.supabase.co';
const SUPABASE_KEY = 'YOUR_ANON_KEY';

// Tables needed in your Supabase project:
// actual_workouts  — main workout records
// set_logs         — embedded sets (FK: workout_id)
// lap_logs         — embedded laps (FK: workout_id)
// pr_ledger        — personal records
// milestones       — milestone events

// The app reads SUPABASE_URL and SUPABASE_KEY from window scope.
// It will attempt to connect on load if both are present.

SQL schema file: add schema.sql to your repo root. A starter schema is generated below based on the Thragg data model.

Starter SQL Schema (copy into Supabase SQL editor)
-- THRAGG PROTOCOL — Supabase Schema v2
-- Run this in your Supabase project SQL editor

create table if not exists actual_workouts (
  id               text primary key,
  planned_workout_id text,
  program_id       text,
  date             date not null,
  session          int default 1,
  type             text not null,  -- strength|run|conditioning|mobility
  pillar           text not null,  -- strong|durable|fast|mobility
  title            text,
  duration         int,
  notes            text,
  vest_weight_lbs  int default 0,
  mile_time_sec    int,
  deviation_from_plan text default 'none',
  sets             jsonb default '[]',
  laps             jsonb default '[]',
  created_at       timestamptz default now()
);

create table if not exists pr_ledger (
  id               text primary key,
  workout_id       text references actual_workouts(id),
  metric           text not null,  -- deadlift|squat|bench|mile_time|murph_time
  value            float not null,
  date             date not null,
  vest_worn        boolean default false,
  previous_pr      float,
  created_at       timestamptz default now()
);

create table if not exists milestones (
  id               text primary key,
  workout_id       text references actual_workouts(id),
  type             text not null,  -- threshold_met|phase_complete|pr_set|streak
  pillar           text,
  label            text not null,
  date             date not null,
  threshold_value  float,
  created_at       timestamptz default now()
);

-- Enable Row Level Security (RLS) — single user, no auth needed for personal use
alter table actual_workouts enable row level security;
alter table pr_ledger       enable row level security;
alter table milestones      enable row level security;

-- Allow all operations from anon key (personal use only)
create policy "allow_all" on actual_workouts for all using (true) with check (true);
create policy "allow_all" on pr_ledger       for all using (true) with check (true);
create policy "allow_all" on milestones      for all using (true) with check (true);
